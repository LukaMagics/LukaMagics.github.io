import queue

class MovingAvg():
    def __init__(self, size):
        self.q = queue.Queue()
        self.size = size
        self.sum = 0

    def nextVal(self, num):
        if self.q.qsize() < self.size:
            self.q.put(num)
            self.sum += num
            avg = self.sum / self.q.qsize()
            return avg
        else:
            self.sum -= self.q.get()
            self.q.put(num)
            self.sum += num
            avg = self.sum / self.q.qsize()
            return avg
    
def main():
    nums = [2,8,19,37,4,5]
    ma = MovingAvg(3)
    results = []
    for num in nums:
        avg = ma.nextVal(num)
        results.append(avg)
    print(results) # [2.0, 5.0, 9.666666666666666, 21.333333333333332, 20.0, 15.333333333333334]
    
if __name__ == "__main__":
    main()

import queue

def josephus(num, target):

    q = queue.Queue()

    for i in range(1, num+1):
        q.put(i)    # 7 1 2 4 5

    jose = []

    while q.qsize() != 0:
        for j in range(target-1):
            q.put(q.get())
        jose.append(q.get())

    return jose

def main():
